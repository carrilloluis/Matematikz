\subsection*{Literatura}
\begin{itemize}
\item{Las ``\texttt{Tradiciones Peruanas}'' es una obra donde se narran :
\begin{tasks}
\task{Únicamente hechos y acciones históricas de los incas.}
\task{Hechos verídico-ficcionales  e históricos de épocas pasadas.}
\task{Costumbres, leyendas y biografías de personajes solamente históricos.}
\task{Crónicas del descubrimiento y la conquista de América.}
\task{Solamente hechos imaginados por Ricardo Palma.}
\end{tasks}
}

\item{La obra ``\texttt{Poemas Humanos}" de César Vallejo trata de :
\begin{tasks}
\task{La Guerra Civil Española donde él participó.}
\task{Sus amores con la dulce y andina Rita.}
\task{El dolor, la desesperanza y la compasión.}
\task{Su identificación con las corrientes literarias de su tiempo.}
\task{Su experiencia literaria en Rusia e Italia.}
\end{tasks}
}
\item{En la obra narrativa de \textit{Julio Ramón Ribeyro} predomina :
\begin{tasks}
\task{El amor en la clase social alta.}
\task{La frustación de la clase media.}
\task{La denuncia contra los poderes dictatoriales.}
\task{La esperanza realizada de sus personajes.}
\task{La justicia como medio de cambio social.}
\end{tasks}
}
\item{Mientras Don Quijote representa las dimensiones idealismo y fantasía, Sancho expresa :
\begin{tasks}
\task{El desprendimiento y el individualismo.}
\task{El derroche y la fidelidad.}
\task{El materialismo y la realidad.}
\task{La ingenuidad y la fiereza.}
\task{La ironía y la templanza.}
\end{tasks}
}
\item{La obra ``\texttt{Crimen y Castigo}" de Fedor Dostoievski es una novela :
\begin{tasks}
\task{Psicológica.}
\task{Costumbrista.}
\task{Romántica.}
\task{Histórica.}
\task{Biográfica.}
\end{tasks}
}
\end{itemize}
