\subsection*{Derivadas e Integrales} % Derivative
\subsubsection*{Derivadas}
\begin{itemize}
\item $\dfrac{d}{dx}\,n = 0$
\item $\dfrac{d}{dx}\,x = 1$ 
\item $\dfrac{d}{dx}\,x^n = n \cdot x^{\,n - 1}$ 
\item $\dfrac{d}{dx}\,\mathrm{e}^x = \mathrm{e}^x$ 
\item $\dfrac{d}{dx}\,\ln\,x = \dfrac1{x}$ 
\item $\dfrac{d}{dx}\,n^{x} = n^{x}\cdot\ln\,x$ 
\end{itemize}
\subsection*{Derivadas de funciones trigonométricas} 
\begin{itemize}
\item $\dfrac{d}{dx}\,\sen\,{x} = \cos\,x$
\item $\dfrac{d}{dx}\,\cos\,{x} = -\sen\,x$

\item $\dfrac{d}{dx}\,\tan\,{x} = \sec^{2}\,x$
\item $\dfrac{d}{dx}\,\cot\,{x} = -\csc^{2}\,x$

\item $\dfrac{d}{dx}\,\sec\,{x} = \sec\,x\cdot\tan\,x$
\item $\dfrac{d}{dx}\,\csc\,{x} = -\csc\,x\cdot\cot\,x$
\end{itemize}
