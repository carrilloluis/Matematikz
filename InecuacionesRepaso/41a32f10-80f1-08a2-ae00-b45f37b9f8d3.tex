\subsection*{Inecuaciones}
\begin{itemize}
\item{Resuelva la siguiente inecuación : $(x^3 + 1) \cdot (x^3 - 1) \cdot (1 - 2x) \cdot x \leq 0$
\begin{tasks}(2)
\task{$\big< -\infty;\,-1\big] \cup \Big[0;\,\dfrac1{2}\Big] \cup \big[1;\,+\infty\big>$}

\task{$\big<-\infty;-1\big] \cup \Big[0;\,\dfrac1{2}\Big] \cup \big[2;\,+\infty\big>$}

\task{$\big< -\infty;\,-1\big] \cup \Big[0;\,\dfrac1{2}\Big] \cup \big[4;\,+\infty\big>$}

\task{$\big< -\infty;\,-1\big] \cup \Big[0;\,\dfrac1{2}\Big] \cup \big[5;\,+\infty\big>$}

\task{$\big< -\infty;\,-1\big] \cup \Big[\dfrac{10}{3};\,\dfrac1{2}\Big] \cup \big]1;\,+\infty\big>$}
\end{tasks}
}
\item{Para que valores de ``\scalebox{1.2}{$a$}'' en la siguiente inecuación cuadrática : $$x^{2} + ax - 2 < 2x^{2} - 2x + 2$$ se cumple para todo $x \in \mathbb{R}$.
\begin{tasks}(3)
\task{$a \in \big<-6;\,2\big>$}
\task{$a \in \big<3;\,6\big>$}
\task{$a \in \big<1;\,3\big>$}
\task{$a \in \big<-10;\,-7\big>$}
\task{$a \in \big<-15;\,-10\big>$}
\end{tasks}
}
\item{Para que valores de ``\scalebox{1.2}{$m$}'' el trinomio : $(m - 2)x^{2} + (4m - 6)x + (5m - 6)$ es positivo $\forall\,x \in \mathbb{R}$.
\begin{tasks}(3)
\task{$\big<-\infty;\,1\big> \cup \big<3;\,+\infty\big>$}
\task{$\big<3;\,+\infty\big>$}
\task{$\big<-\infty;\,1\big>$}
\task{$\big<-1;\,3\big>$}
\task{$\big<-1;\,+\infty\big>$}
\end{tasks}
}
\item{Si el {\tt conjunto solución} de la inecuación : $$\dfrac{(x^{3} - 8)(x^{2} - 9)(x^{2} - 7x + 13)}{(1 - x)(4 - x^{2})} \leq 0$$ es : $\big<-a;\,b\big> \cup \big\{-c;\,c\big\};\ c < 0$. Determinar el valor de ``\scalebox{1.2}{$a + b + c$}''.
\begin{tasks}(5)
\task{-3}
\task{-1}
\task{0}
\task{1}
\task{3}
\end{tasks}
}
\item{Sabiendo que la siguiente ecuación cuadrática : $ax^{2} + (1 - 2a)x + a \geq 0; \forall\,x \in \mathbb{R}$, el conjunto de los valores de ``\scalebox{1.2}{$a$}'' es :
\begin{tasks}(3)
\task{$\big]-\infty;\,2\big[$}
\task{$\Big[\,\dfrac1{4};\,+\infty\Big[$}
\task{$\big[2;\,3\big]$}
\task{$\big[1;\,+\infty\big[$}
\task{$\Big]\dfrac1{4};\,+\infty\Big[$}
\end{tasks}
}
\item{Sean ``\scalebox{1.2}{$a$}'' y ``\scalebox{1.2}{$b$}'' números reales positivos tales que : ``\scalebox{1.2}{$a \cdot b = 8$}''. Hallar el menor valor de ``\,\scalebox{1.2}{$a + b$}\,''.
\begin{tasks}(5)
\task{$\dfrac1{2}$}
\task{$1$}
\task{$2$}
\task{$4 \sqrt{2}$}
\task{$2 \sqrt{2}$}
\end{tasks}
}
\end{itemize}
