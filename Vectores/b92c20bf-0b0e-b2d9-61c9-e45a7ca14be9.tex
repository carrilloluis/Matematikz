Dado \quad$\overrightarrow{a\,} = (3,\,1,\,-1)$\quad y \quad$\overrightarrow{b\,} = (2,\,3,\,4)$
\begin{itemize}
\item{Calcular los módulos de $\overrightarrow{a\,}$\;\;y de\;\;$\overrightarrow{b\,}$\qquad ó \qquad$\lvert\mid\vec{a}\mid\rvert$\;\;y\;\;$\lvert\mid\vec{b}\mid\rvert$

\begin{eqnarray*}
\vec{a} &=& (3,\,1,\,-1) \\
\lvert\mid\vec{a}\mid\rvert &=& \sqrt{(3)^{2} + (1)^{2} + (-1)^{2} \,} \\
||\;{\vec{a}}\;|| &=& \sqrt{9 + 1 + 1\;} = \framebox{$\sqrt{11\,}$}
\end{eqnarray*}
\begin{eqnarray*}
\vec{b} &=& (2,\,3,\,4) \\
||\;{\vec{b}}\;|| &=& \sqrt{(2)^{2} + (3)^{2} + (4)^{2} \,} \\
||\;{\vec{b}}\;|| &=& \sqrt{4 + 9 + 16\;} = \framebox{$\sqrt{29\,}$}
\end{eqnarray*}
}
\item{Hallar el producto vectorial entre $\overrightarrow{a\,}$\;\;y\;\;$\overrightarrow{b\,}$
\begin{eqnarray*}
\vec{a} \times \vec{b} &=& \begin{vmatrix} \vec{i} & \vec{j} & \vec{k} \\ 3 & 1 & -1 \\ 2 & 3 & 4 \end{vmatrix} \\
\vec{a} \times \vec{b} &=& +\,\vec{i}\;\;\begin{vmatrix} 1 & -1 \\ 3 & 4 \end{vmatrix} - \vec{j}\;\;\begin{vmatrix} 3 & -1 \\ 2 & 4 \end{vmatrix} + \vec{z}\;\;\begin{vmatrix} 3 & 1 \\ 2 & 3 \end{vmatrix} \\
\vec{a} \times \vec{b} &=& 7\,\vec{i} - 14\,\vec{j} + 7\,\vec{k} \\
\vec{a} \times \vec{b} &=& 7 \cdot (1,\,-2,\,1)
\end{eqnarray*}
}
\item{Hallar el vector unitario ortogonal entre $\overrightarrow{a\,}$\;\;y\;\;$\overrightarrow{b\,}$\quad ó\quad$\vec{u}^{\,\perp} = \dfrac{\vec{a} \times \vec{b}}{||\,\vec{a} \times \vec{b}\,||}$
\begin{eqnarray*}
\vec{a} \times \vec{b} &=& (1,\,-2,\,1) \\
||\,\vec{a} \times \vec{b}\,|| &=& \sqrt{(1)^{\,2} + (-2)^{\,2} + (1)^{\,2}\,} \\
||\,\vec{a} \times \vec{b}\,|| &=& \sqrt{1 + 4 + 1\,} = \sqrt{6\,} \\
\vec{u}^{\,\perp} &=& \dfrac{\vec{a} \times \vec{b}}{||\,\vec{a} \times \vec{b}\,||} = \dfrac{(1,\,-2,\,1)}{\sqrt{6}} \\
 \vec{u}^{\,\perp} &=& \Big(\dfrac{1}{\sqrt{6}}, \,\dfrac{-2}{\sqrt{6}}, \,\dfrac{1}{\sqrt{6}}\Big)
\end{eqnarray*}
}
\item{Determinar el área del paralelogramo que tiene por lados los vectores $\overrightarrow{a\,}$\;\;y\;\;$\overrightarrow{b\,}$
\begin{eqnarray*}
\mid\vec{a} \times \vec{b}\mid &=& \mid\sqrt{(1)^{\,2} + (-2)^{\,2} + (1)^{\,2}}\mid\;= \sqrt{6\,}\,u^{\,2}
\end{eqnarray*}
}
\end{itemize}
