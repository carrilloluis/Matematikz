\subsection*{Factorización}
\begin{itemize}
\item{Si $\;\;F(x) = x - 2\quad$ es factor primo del polinomio : \\ $P(x) = ax^3 - 2ax^2 - bx + 2\;;\; ab \neq 0$. Calcular el valor de $b$.
\begin{tasks}(5)
\task{-1}
\task{0}
\task{$\frac1{2}$}
\task{1}
\task{-2}
\end{tasks}
}
\item{Factorizar : $M(x;\,y;\,z) = x^2y + xy^2 + 5x^2yz + 5xy^2z$, e indicar un factor primo.
\begin{tasks}(5)
\task{$x^2$}
\task{$xy$}
\task{$1 + 5x$}
\task{$x + y$}
\task{$5z$}
\end{tasks}
}
\item{Indicar la suma de factores no comunes de los polinomios : \\ $P(x) = 3x^2 - 16x + 21\qquad\qquad Q(x) = 5x^2 - 14x - 3$
\begin{tasks}(5)
\task{$5x + 1$}
\task{$3x - 7$}
\task{$8x - 1$}
\task{$8x - 5$}
\task{$8x - 6$}
\end{tasks}
}
\item{Hallar la suma de los factores primos del polinomio : \\ $P(x) = x^4 - 10x^2 + 9$.
\begin{tasks}(5)
\task{$4x$}
\task{$4x + 4$}
\task{$4x + 1$}
\task{$4x - 1$}
\task{$0$}
\end{tasks}
}
\item{Si $(x^2 + ax + b)$ es un factor primo del polinomio : \\ $P(x) = (x^2 + x - 4)^2 - 2x^2 - 2x$, el valor de $ab$ es:
\begin{tasks}(5)
\task{4}
\task{-2}
\task{2}
\task{-4}
\task{-8}
\end{tasks}
}
\item{Si $\;Q(x) = ax^2 + bx + 1\;$ es el factor primo común de los polinomios : \\ $P(x) = (x^2 - 1)(x^2 - 5) - (3x + 1)$ \\ $Q(x) = 3x^4 + 2x^3 - 2x^2 + 3x - 2\,;\qquad$ Calcular el valor de $Q(2)$.
\begin{tasks}(5)
\task{-2}
\task{1}
\task{3}
\task{5}
\task{10}
\end{tasks}
}
\item{Un factor de : $\quad P(c) = ac^2 + bc - a^2c - ab\,;\qquad$ es :
\begin{tasks}(3)
\task{$ac - b$}
\task{$c + a$}
\task{$c - a$}
\task{$c + b$}
\task{$ac + b + 1$}
\end{tasks}
}
\item{Indicar el número de factores primos en : $P(x) = x ^2 + 4 + (x + 1)^2 - (x - 1)^2$
\begin{tasks}(5)
\task{0}
\task{1}
\task{2}
\task{3}
\task{4}
\end{tasks}
}
\item{Indicar el número de factores primos del polinomio : \\ $P(a; b) = [(a + b + ab)^2 - 4ab\,(a + b)]$
\begin{tasks}(5)
\task{1}
\task{2}
\task{3}
\task{4}
\task{5}
\end{tasks}
}
\item{Determinar el factor primo común de : \\ $P(x) = 2x^2 + x - 3\,;\qquad\qquad Q(x) = 6x^2 + 7x - 3$
\begin{tasks}(5)
\task{$x - 1$}
\task{$2x - 3$}
\task{$x + 5$}
\task{$3x - 1$}
\task{$2x + 3$}
\end{tasks}
}
\item{Indicar el número de factores primos de : $P(a;\,b) = a^4b^3 + 8a^3b^4 + 15a^2b^5$
\begin{tasks}(5)
\task{1}
\task{2}
\task{3}
\task{4}
\task{5}
\end{tasks}
}
\item{Factorizar el polinomio : $P(x;\,y) = x^2 + 4xy +3y^2 + 5x + 7y + 4$, luego seleccionar un factor primo.
\begin{tasks}(3)
\task{$x + y - 1$}
\task{$x + y + 4$}
\task{$x + 3y + 1$}
\task{$x + y + 1$}
\task{$x + 5$}
\end{tasks}
}
\item{Factorizar el polinomio : $P(x;\,y) = 2x^2 - 5xy - 12y^2 - 6y + 7x + 6$, e indicar la suma de coeficientes de uno de los factores primos.
\begin{tasks}(5)
\task{-2}
\task{-1}
\task{0}
\task{3}
\task{5}
\end{tasks}
}
\item{Factorizar el polinomio : $P(x) = x^4 + 8x^3 + 21x^2 + 22x + 8$, e indicar el número de factores primos.
\begin{tasks}(5)
\task{0}
\task{1}
\task{2}
\task{3}
\task{4}
\end{tasks}
}
\item{Factorizar el polinomio : $P(x) = x^4 - 2x^3 + 5x^2 + 4x + 7$, e indicar la suma de coeficientes de un factor primo.
\begin{tasks}(5)
\task{-3}
\task{-1}
\task{0}
\task{2}
\task{3}
\end{tasks}
}
\end{itemize}
