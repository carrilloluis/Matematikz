\begin{tikzpicture}
	\coordinate (O) at (0.0, 0.0);
	% A = (size * Cos 90, size * Sen 90
	\coordinate (A) at (.0, 4);
	% (90 - [360 / 3]) -> (90 + 120 = 210) -> 210 - 180 = 30
	% B = (- size * Cos 30, - size * Sen 30 
	\coordinate (B) at (-1.7320, 3);
	\coordinate (C) at (-1.7320, -1);
	\coordinate (D) at (1.7320, -1);
	\coordinate (E) at (1.7320, 3);

	\begin{scope}[line width=.5mm]
	\draw (A) -- (B) -- (E) -- cycle;
	\draw (B) -- (C) -- (D) -- (E); 
	\end{scope}

	\begin{scope}[dashed, gray]
	\draw (A) -- (O);
	\draw (C) -- (O);
	\draw (D) -- (O);
	\end{scope}
