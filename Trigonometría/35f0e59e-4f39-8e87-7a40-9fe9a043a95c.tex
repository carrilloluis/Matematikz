\subsection*{Identidades Trigonométricas}

\subsubsection*{Pitagóricas}
\begin{itemize}
\item{\scalebox{1.3}{$\sen^{\,2} a + \cos^{\,2} a = 1$}}
\item{\scalebox{1.3}{$1 + \tan^{\,2} a = \sec^{\,2} a$}}
\item{\scalebox{1.3}{$1 + \cot^{\,2} a = \csc^{\,2} a$}}
\end{itemize}

\subsubsection*{Razón o cociente}
\begin{itemize}
\item{\scalebox{1.3}{$\tan \,\theta = \dfrac{\sen \,\theta}{\cos \,\theta}$}}
\item{\scalebox{1.3}{$\cot \,\theta = \dfrac{\cos \,\theta}{\sen \,\theta}$}}
\end{itemize}

\subsubsection*{Recíprocas}
\begin{itemize}
\item{\scalebox{1.3}{$\sen \,\alpha \cdot \csc \,\alpha = 1$}}
\item{\scalebox{1.3}{$\cos \,\alpha \cdot \sec \,\alpha = 1$}}
\item{\scalebox{1.3}{$\tan \,\alpha \cdot \cot \,\alpha = 1$}}
\end{itemize}
