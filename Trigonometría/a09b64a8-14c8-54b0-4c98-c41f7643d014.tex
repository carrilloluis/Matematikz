\subsection*{Funciones trigonométricas de la suma y diferencia de 2 arcos}

\begin{itemize}
\item{\scalebox{1.3}{$\sen \,(\alpha \pm \theta) = \sen \alpha \cdot \cos \theta \pm \sen \theta \cdot \cos \alpha$}}
\item{\scalebox{1.3}{$\cos \,(\alpha \pm \theta) = \cos \alpha \cdot \cos \theta \mp \sen \alpha \cdot \sen \theta$}}
\item{\scalebox{1.3}{$\tan \,(\alpha \pm \theta) = \dfrac{\tan \alpha \pm \tan \theta}{1 \mp \tan \alpha \cdot \tan \theta}$}}
\item{\scalebox{1.3}{$\cot \,(\alpha \pm \theta) = \dfrac{\cot \alpha \cdot \cot \theta \mp 1}{\cot \theta \pm \cot \alpha}$}}
\end{itemize}
