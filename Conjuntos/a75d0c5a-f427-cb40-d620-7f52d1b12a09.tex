\begin{enumerate}[I.]
\item{$B - A \neq \varnothing$}
\item{$\varnothing \in (A - B)$}
\item{$A \cap A = \{\varnothing\}$}
\end{enumerate}

\begin{tasks}(3)
\task{Solo I}\task{Solo II}\task{Solo III}\task{I y II}\task{I y III}

\begin{tasks}(5)
\task{1}\task{3}\task{5}\task{7}\task{9}

\item{

\begin{enumerate}[I.]

\item{$n(A) \neq n(B)\;\vee\;n(C) = n(D)$}
