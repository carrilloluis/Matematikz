\item{Si: $A = \{\varnothing\,;\,a\,;\,\{a\}\,;\,\{a\,;\,b\}\,;\,\{\varnothing\}\}$. Indicar las proposiciones verdaderas.

\begin{enumerate}[I.]
\item{$a \subset A\quad \wedge\quad \{a\,;\,b\} \subset A$}
\item{$\{\varnothing\} \notin A\quad \vee\quad \{\varnothing\} \subset A$}
\item{$\varnothing \subset A\quad \wedge\quad \varnothing \in A$}
\end{enumerate}

$A = \{x \in \mathbb{N}\;/\;2x \leq 13\}$ \\
$B = \{x \in A\;/\;(x^{\,2} - 2x) \notin A\}$ \\

\begin{enumerate}[I.]
\item{$\exists\,x \in A\;/\;x^{\,2} - 5 > 4$}
\item{$\forall\,x \in (A - B)\;/\;2x + 5 < 8$}
\item{$\exists\,x \in (A - B)\;/\;x^{\,2} \in B$}
\end{enumerate}
