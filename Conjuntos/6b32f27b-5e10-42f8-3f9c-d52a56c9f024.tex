\item{Sea el conjunto \quad$B = \big\{n;\,\{n\};\,\varnothing;\,7\big\}$ \\ Hallar el valor de verdad de las siguientes proposiciones :

\begin{enumerate}[I.]
\item{$\{n\} \in B \wedge \varnothing \in B$}
\item{$\varnothing \in B \vee \{\varnothing\} \subset B$}
\item{$\{\varnothing;\,n\} \subset B \implies \{n; \,\{n\}\} \in B$}
\end{enumerate}

\begin{enumerate}[I.]
