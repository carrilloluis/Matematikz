\item{Determine la \textit{cuarta proporcional} de $4$, $10$ y $8$.
	\begin{tasks}(5)
		\task{20}
		\task{25}
		\task{10}
		\task{8}
		\task{30}
	\end{tasks}
	\noindent \textbf{Soluci√≥n} -- {\small Sean $a = 4$, $b = 10$ y $c = 8$, la \textit{cuarta proporcional} es $d$ en} :
	\begin{eqnarray*}
		\dfrac{a}{b} &=& \dfrac{c}{d} \\
		\dfrac{4}{10} &=& \dfrac{8}{d} \\
		d &=& \dfrac{8 \cdot 10}{4} \\
		d &=& \dfrac{2^4 \cdot 5}{2^2} = \fbox{20} 
	\end{eqnarray*}
