		\noindent \textbf{Solución} -- {\small Sean $a = 7$ y $b = 23$, la \textit{tercera diferencial} es $c$ en} :
		\begin{eqnarray*}
			a - b &=& b - c \\
			7 - 23 &=& 23 - c \\
			7 - 23 - 23 &=& - c \\
			7 - 46 &=& - c \\
			- 39 &=& - c \\
			c &=& \fbox{39}
		\end{eqnarray*}
