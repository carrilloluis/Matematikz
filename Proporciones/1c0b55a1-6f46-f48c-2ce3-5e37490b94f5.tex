\item{Determine la \textit{tercera proporcional} de $25$ y $45$
	\begin{tasks}(5)
		\task{35}
		\task{18}
		\task{9}
		\task{25}
		\task{81}
	\end{tasks}
	\noindent \textbf{Soluci√≥n} -- {\small Sean $a = 25$ y $b = 45$, la \textit{tercera proporcional} es $c$ en} :
	\begin{eqnarray*}
		\dfrac{a}{b} &=& \dfrac{b}{c} \\ 
		\dfrac{25}{45} &=& \dfrac{45}{c} \\ 
		c &=& \dfrac{45 \cdot 45}{25} \\
		c &=& \dfrac{5^2 \cdot 3^4}{5^2} \\
		c &=& 3^4  = \fbox{81} 
	\end{eqnarray*}
