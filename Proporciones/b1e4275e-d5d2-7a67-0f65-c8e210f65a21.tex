	\noindent \textbf{Soluci√≥n} -- {\small Sean $a = 4$, $c = 16$, la \textit{media proporcional} es $b$ en} :
	\begin{eqnarray*}
		\dfrac{a}{b} &=& \dfrac{b}{c} \\
		\dfrac{4}{b} &=& \dfrac{b}{16} \\
		b^{\,2} &=& 4 \cdot 16 = 2^2 \cdot 2^4 \\

	\end{eqnarray*}
