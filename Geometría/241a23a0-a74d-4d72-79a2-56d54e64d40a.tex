\begin{eqnarray*}
\text{Sea}\quad d &=& \sqrt{\,(x_{2} - x_{1})^{\,2} + (y_{2} - y_{1})^{\,2}\,} \\
d_{A, B} &=& \sqrt{ (3 - 0)^{\,2} + (4 - 0)^{\,2}\,} \\
d_{A, B} &=& \sqrt{ 3^{\,2} + 4^{\,2}\,} \\
d_{A, B} &=& \sqrt{\,9 + 16\,} = \sqrt{\,25\,}\\
d_{A, B} &=& \pm\, 5
\end{eqnarray*}
\begin{eqnarray*}
d_{B, C} &=& \sqrt{ (6 - 3)^{\,2} + (0 - 4)^{\,2}\,} \\
d_{B, C} &=& \sqrt{ (3)^{\,2} + (-4)^{\,2}\,} \\
d_{B, C} &=& \sqrt{\,9 + 16\,} = \sqrt{\,25\,} \\
d_{B, C} &=& \pm\, 5
\end{eqnarray*}
\begin{eqnarray*}
d_{A, C} &=& \sqrt{ (6 - 0)^{\,2} + (0 - 0)^{\,2}\,} \\
d_{A, C} &=& \sqrt{ (6)^{\,2} + (0)^{\,2}\,} \\
d_{A, C} &=& \sqrt{\,36 + 0\,} = \sqrt{\,36\,} \\
d_{A, C} &=& \pm\, 6
\end{eqnarray*}
$\therefore \quad d_{A,B} = d_{B,C} \neq d_{A,C} \implies \triangle ABC\quad \text{es Isósceles}$
