\subsection*{Inecuaciones}
\begin{itemize}
\item{Resolver : \scalebox{1.15}{$4x^2 - 24x + 36 > 0$}
\begin{tasks}(3)
\task{$\big[2,\,6\big]$}
\task{$\big[ -6,\,-1 \big[$}
\task{$\mathbb{R} - {3}$}
\task{$\{3\}$}
\task{$\emptyset$}
\end{tasks}
}
\item{La solución de la inecuación : \scalebox{1.15}{$3x^2 - 2x < 8$}\,; es un intervalo finito. La suma algebraica de los extremos de dicho intervalo es :
\begin{tasks}(5)
\task{$-\dfrac{4}{3}$}
\task{$-\dfrac{2}{3}$}
\task{$\dfrac{4}{3}$}
\task{$\dfrac{2}{3}$}
\task{$\dfrac1{3}$}
\end{tasks}
}
\item{Resolver : \scalebox{1.15}{$\dfrac1{x - 2} \leq \dfrac{2}{x - 4}$}
\begin{tasks}(3)
\task{$\big[2,\,4\big]$}
\task{$\big[0,\,2\big> \cup \big[4,\,+\infty\big>$}
\task{$\big[3,\,4\big]$}
\task{$\big[-4,\,0\big[$}
\task{$\emptyset$}
\end{tasks}
}
\item{La solución de la inecuación es : \qquad\scalebox{1.15}{$x^2 - 9 < 4$}

\begin{tasks}(2)
\task{$\big[-6,\,-3\big] \cup \big[-3,\,3\big]$}
\task{$\big[-3,\,3\big]$}
\task{$\big[-5,\,5\big]$}
\task{$\big[-3,\,3\big] \cup \big<3,\,5\big]$}
\task{$\big<-5,\,-3\big] \cup \big[3,\,5\big]$}
\end{tasks}
}

\item{Dar el conjunto solución : \qquad\scalebox{1.15}{$x^4 - 13x^2 + 36 < 0$}
\begin{tasks}(3)
\task{$[-3, -2]$}
\task{$[-3, -2] \cup [2, 3]$}
\task{$[2, 3]$}
\task{$\mathbb{R}$}
\task{$\big[-3,\,3\big]$}
\end{tasks}
}
\item{Dar el conjunto solución : \scalebox{}{$\dfrac{x - 1}{x + 4} \leq \dfrac{x - 3}{x - 2}$}
\begin{tasks}(3)
\task{$\big]-4,\, 2\big[$}
\task{$[-4, 2] \cup [\dfrac{7}{2},\,\infty[$}
\task{$[\dfrac{7}{2}, +\infty>$}
\task{$[-4, +\infty [$}
\task{$[-4, 2]$}
\end{tasks}
}
\item{Dar el conjunto solución : $(x^2 + 2x - 3)(3x - 4 - x^2) \geq 0$
\begin{tasks}(3)
\task{$\mathbb{R}$}
\task{$]-\infty,\,3]$}
\task{$[1 +\infty[$}
\task{$[-3, 1]$}
\task{$[1, +\infty[$}
\end{tasks}
}
\item{Resolver : $\dfrac{(x + 3)(x + 2)(x + 1)}{(x + 5)(x - 3)} \geq \dfrac{(x + 3)(x + 2)}{(x - 3)}$
\begin{tasks}(3)
\task{$[-5, -3] \cup [-2, 3]$}
\task{$[-2, 3]$}
\task{$]-3, 3]$}
\task{$[-5, 3]$}
\task{$[-2, 3[$}
\end{tasks}
}
\item{Resolver : $9^x + 9 \leq {10{.}3}^x$ e indicar su intervalo solución : 
\begin{tasks}(3)
\task{$[0,\,\dfrac1{2}]$}
\task{$[0,\,2]$}
\task{$[0,\,2[$}
\task{$]0,\,2[$}
\task{$]0,\,\dfrac1{2}]$}
\end{tasks}
}
\item{Resolver el siguiente sistema de inecuaciones para valores \\ enteros y positivos de ``\scalebox{1.15}{$x$}'' : \\ \scalebox{1.3}{$2^{\,3x - 5} > 4^{\,2x - 4};\qquad\sqrt[3]{3^{\frac{5x + 1}{2}}} < 9^{\frac{3(x + 1))}{5}}$}
\begin{tasks}(5)
\task{1}
\task{2}
\task{3}
\task{4}
\task{5}
\end{tasks}
}
\item{Resolver : $\dfrac{3x^2 + 7x - 6}{x^2 - x - 6} > \dfrac{3x^2 + 16x - 12}{x^2 - 4x - 12}$
\begin{tasks}(2)
\task{$]-\infty,\,2]$}
\task{$[0,\,\dfrac{2}{3}]$}
\task{$[3,\,6]$}
\task{$[-2,\,\dfrac{2}{3}]$}
\task{$[-\infty,\,2] \cup \big[0,\,\dfrac{2}{3}\big] \cup [3,\,6]$}
\end{tasks}
}
\item{Resolver : \quad\scalebox{1.15}{$x^2 - 16 < 3$}
\begin{tasks}(3)
\task{$\big[-5,\,-4\big] \cup [4,\,5]$}
\task{$[-4,\,4] \cup [3,5]$}
\task{$[-4,\,4]$}
\task{$[-5,\,5]$}
\task{$[-5,\,-4]$}
\end{tasks}
}
\item{Resolver : \scalebox{1.15}{$\dfrac{(x + 3)^5 (x - 4)^3 (x + 5 )^4 (x - 6)}{x^2 (x + 1)^7} \leq 0$}\,; Indicar la suma de los valores enteros del conjunto solución.
\begin{tasks}(5)
\task{3}
\task{5}
\task{7}
\task{8}
\task{10}
\end{tasks}
}
\item{Resolver : $(2x - 3)^{26}(2x - 5)^{36}(x - 9)^{3556}(x - 7)^{2000} > 0$
\begin{tasks}(2)
\task{$x \in \mathbb{R} - \{\dfrac{3}{2}\} : \{\dfrac{5}{2}\} : \{7\} : \{9\} $}
\task{$x \in \mathbb{R}$}
\task{$x \in \mathbb{R} - \{\dfrac{3}{2}\}$}
\task{$x \in \mathbb{R} - \{\dfrac{5}{2}\}$}
\task{$x \in \mathbb{R} - \{7\}$}
\end{tasks}
}
\end{itemize}
