\subsection*{Ley de los exponentes y radicales}

\paragraph*{Exponentes}
\begin{itemize}
\item{$x^n \cdot x^m = x^{m + n}$}
\item{$(x^a)^b = x^{a \cdot b}$}
\item{$\dfrac1{x^{a}} = x^{-a}$}
\end{itemize}

\paragraph*{Radicales}
\begin{itemize}
\item{$\sqrt{a} = a^{\frac1{2}}$}
\item{$\sqrt[3]{a} = a^{\frac1{3}}$}
\end{itemize}
