\subsection*{Identidades notables.}
\hrule
\vspace{6pt}
\begin{itemize}
\item{$\mathrm{K}(a,b,c) = a^{3}b^{2} + b^{3}c^{2}  - a^{3}c^{2} - b^{5}$}
\item{$\mathrm{M}(x,y) = x^{4} + xy^{3} + x^{3}y + y^{4}$}
\item{$\mathrm{P}(x,y) = 4x^{2}y^{2} - (x^{2}+ y^{2} - 25)^{2}$}
\item{$\mathrm{F}(x,y) = x^{4} + 3x^{3}y + 3x^{2}y^{2} + xy^{3}$}
\item{$\mathrm{P}(x,a) = (1 + ax)^{2} - (a + x)^{2} - a^{2} + 1$}
% duplicado
\item{$\mathrm{G}(x,y,z) = x^{3}y^{2} + y^{3}z^{2} - x^{3}z^{2} - y^{5}$}
\item{$\mathrm{M}(a) = 2a^{4} + 3a^{3} + 2a + 3$}
\item{$\mathrm{A}(x,y,z) = x^{2} + y^{2} + x(y + z) + y (x + z)$}
\item{$\mathrm{A}(x,y,z) = x^{2}y^{2} + x(y + z) + y (x + z)$}
\item{$\mathrm{P}(x,y) = x^{3} + 4x^{2} + 4x - xy^{2}$}
\item{$\mathrm{S}(x,y) = (x^{2} + y^{2} - 4c^{2})^{2} - 4x^{2}y^{2}$}
\item{$\mathrm{L}(m,n) = 27m^{3} + 27m^{2}n + 9mn^{2} - 7n^{3}$}
\end{itemize}
