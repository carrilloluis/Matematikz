\subsection*{Factor Común y/o Agrupación de términos.}
\hrule
\vspace{6pt}
\begin{itemize}
\item{$\mathrm{F}(x) = (x + 2)(x + 3)(x + 4) + (x + 3)(x + 4) - (x + 4)$}
\item{$\mathrm{M}(y) = (y + 1)(2y + 1)(3y + 1) + (y +1 )^2 + y + y^2$}
\item{$\mathrm{A}(a, b, x, y) = (x - y)(a - b) - (x + y)(b - a) + b - a$}
\item{$\mathrm{P}(a, b, c) = a(b^2 + c^2) + b(c^2 + a^2)$}
\item{$\mathrm{A}(x, y) = x(x^2 - y^2 + xz) - y^{2}z$}
\item{$\mathrm{A}(x) = x^3 - ax^2 - bx^2 + abx - cx^2 + acx + bcx - abc$}
\item{$\mathrm{F}(a,b,c) = bda^2 + bdc^2 + a^2bc + bcd^2 + adc^2 + adb^2 + ab^2c + acd^2$}
\item{$\mathrm{P}(x, y) = 4x^{m+2p}y^{n} + 8x^{m+p}y^{n+q} + 4x^{m}y^{n+2q}$}
\item{$\mathrm{F}(x) = (x^2 - 3ax)^2 + ax(2x^2 - 6ax)$}
\item{$\mathrm{H}(x) = (x^2 - 1)(x + 2)(x + 3) + (x^2 - 1)(x + 2) + x^2 - 1$}
\end{itemize}
