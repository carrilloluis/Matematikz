\subsection*{Matrices}

\subsubsection*{Multiplicación por un escalar}
$$
2 \cdot \begin{pmatrix} 1 & -3 \\ 7 & 2 \end{pmatrix} = \begin{pmatrix} 2 & -6 \\ 14 & 4 \end{pmatrix}
$$

\subsubsection*{Adición de matrices}
\[
\begin{pmatrix} 1 & 2 \\ 5 & 1\end{pmatrix} + \begin{pmatrix} 3 & -1 \\ 2 & 1 \end{pmatrix} = \begin{pmatrix} 4 & 1 \\ 7 & 2 \end{pmatrix}
\]

\subsubsection*{Multiplicación entre matrices}
\[
\begin{pmatrix} 3 & -2 \\ -1 & 4\end{pmatrix} \times \begin{pmatrix} 3 & 1 \\ 2 & 4 \end{pmatrix} = \begin{pmatrix} 9 + (-4) & 3 + (-8) \\ -3 + 8 & -1 + 16 \end{pmatrix} = \begin{pmatrix} 5 & -5 \\ 5 & 15 \end{pmatrix}
\]
\subsubsection*{Expresar sistemas de ecuaciones como matrices}
\begin{eqnarray*}
3x - 2y &=& 1 \\
-x + 4y &=& 3
\end{eqnarray*}
\[
\begin{pmatrix} 3 & -2 \\ -1 & 4\end{pmatrix} \cdot \begin{pmatrix} x \\ y \end{pmatrix} = \begin{pmatrix} 1 \\ 3 \end{pmatrix}
\]
